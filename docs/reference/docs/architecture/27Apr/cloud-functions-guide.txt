      applicationId, // Store parent application ID for querying
      eventTimestamp,
      eventType,

    // Store the event

    logger.info(`Stored metadata event ${eventId} for application ${applicationId}`); // Add logging

      dropOffTime.setSeconds(dropOffTime.getSeconds() + 30); // 30 seconds from now [cite: 379]

        .collection('backgrounded_apps')
        .doc(applicationId)
        .set({
          applicationId,
          backgroundedAt: eventTimestamp, // Use the timestamp from the event
          dropOffCheckTime: admin.firestore.Timestamp.fromDate(dropOffTime), // Store as Firestore Timestamp
          processed: false
        }, { merge: true }); // Use merge to update if exists
        logger.info(`Scheduled drop-off check for ${applicationId}`); // Add logging [cite: 380]
    }

    // If it's a foreground event, cancel any pending drop-off check
    if (eventType === 'APP_FOREGROUNDED') { [cite: 380]

            .collection('backgrounded_apps')
            .doc(applicationId)
